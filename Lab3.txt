using System;

// Оголошення класу Person
public class Person
{
    protected int age; // Змінився модифікатор доступу з private на protected

    // Метод встановлення віку
    public void ВстановитиВік(int вік)
    {
        age = вік;
    }

    // Метод привітання
    public void Привітатися()
    {
        Console.WriteLine("Привіт!");
    }
}

// Клас Student, який успадковується від класу Person
public class Student : Person
{
    // Метод навчання
    public void Навчатися()
    {
        Console.WriteLine("Я вчуся");
    }

    // Метод відображення віку
    public void ПоказатиВік()
    {
        Console.WriteLine("Мій вік: " + age + " років");
    }
}

// Клас Professor, який успадковується від класу Person
public class Professor : Person
{
    // Метод пояснення
    public void Пояснювати()
    {
        Console.WriteLine("Я пояснюю");
    }
}

// Тестовий клас
public class ТестСтудентПрофесор
{
    // Метод Main для виконання тестування
    public static void Main(string[] args)
    {
        // 1. Створення нової людини і змушення її привітатися
        Person людина = new Person();
        людина.Привітатися();

        // 2. Створення нового студента, встановлення віку, привітання та відображення віку
        Student студент = new Student();
        студент.ВстановитиВік(20); // Припустимо, що вік студента - 20 років
        студент.Привітатися();
        студент.ПоказатиВік();

        // 3. Створення нового професора, встановлення віку, привітання та початок пояснення
        Professor професор = new Professor();
        професор.ВстановитиВік(45); // Припустимо, що вік професора - 45 років
        професор.Привітатися();
        професор.Пояснювати();
    }
}
